70 /******************************************************************************
71  * NOTE 1: This project provides two demo applications. A simple blinky style
72  * project, and a more comprehensive test and demo application. The
73  * mainCREATE_SIMPLE_BLINKY_DEMO_ONLY setting in main.c is used to select
74  * between the two. See the notes on using mainCREATE_SIMPLE_BLINKY_DEMO_ONLY
75  * in main.c. This file implements the comprehensive test and demo version.
76  *
77  * NOTE 2: This file only contains the source code that is specific to the
78  * full demo. Generic functions, such FreeRTOS hook functions, and functions
79  * required to configure the hardware, are defined in main.c.
80  ******************************************************************************
81  *
82  * main_full() creates a set of standard demo tasks (including a set of tasks
83  * that test the interrupt nesting behaviour), some application specific tasks,
84  * and a timer. It then starts the scheduler. The web documentation provides
85  * more details of the standard demo application tasks, which provide no
86  * particular functionality, but do provide a good example of how to use the
87  * FreeRTOS API.
88  *
89  * The interrupt nesting test tasks require that two timers are configured to
90  * generate interrupts. The interrupt service routines are defined in
91  * IntQueueTimer.c, and can be used as examples for application writers. They
92  * do not, however, directly demonstrate the use of FreeRTOS safe API functions
93  * (those that end in "FromISR"). Therefore, a dummy interrupt implementation
94  * called Dummy_IRQHandler() is provided at the end of main.c.
95  *
96  * In addition to the standard demo tasks, the following tasks and timer are
97  * defined and/or created within this file:
98  *
99  * "Reg test" tasks - These fill the registers with known values, then check
100  * that each register maintains its expected value for the lifetime of the
101  * task. Each task uses a different set of values. The reg test tasks execute
102  * with a very low priority, so get preempted very frequently. A register
103  * containing an unexpected value is indicative of an error in the context
104  * switching mechanism.
105  *
106  * "Check" software timer - The check timer period is initially set to three
107  * seconds. Its callback function checks that all the standard demo tasks, and
108  * the register check tasks, are not only still executing, but are executing
109  * without reporting any errors. If the check timer callback discovers that a
110  * task has either stalled, or reported an error, then it changes the period of
111  * the check timer from the initial three seconds, to just 200ms. The callback
112  * function also toggles the LED each time it is called. This provides a visual
113  * indication of the system status: If the LED toggles every three seconds,
114  * then no issues have been discovered. If the LED toggles every 200ms, then
115  * an issue has been discovered with at least one task.
116  */

181 void main_full( void )
182 {
183 TimerHandle_t xCheckTimer = NULL;
184 /* The register test tasks are asm functions that don't use a stack. The
185 stack allocated just has to be large enough to hold the task context, and
186 for the additional required for the stack overflow checking to work (if
187 configured). */
188 const size_t xRegTestStackSize = 25U;
189 
190  /* Create the standard demo tasks, including the interrupt nesting test
191  tasks. */
192  vStartInterruptQueueTasks();
193  vCreateBlockTimeTasks();
194  vStartCountingSemaphoreTasks();
195  vStartRecursiveMutexTasks();
196 
197  /* Create the register test tasks as described at the top of this file.
198  These are naked functions that don't use any stack. A stack still has
199  to be allocated to hold the task context. */
200  xTaskCreate( vRegTest1Task, /* Function that implements the task. */
201  "Reg1", /* Text name of the task. */
202  xRegTestStackSize, /* Stack allocated to the task. */
203  NULL, /* The task parameter is not used. */
204  tskIDLE_PRIORITY, /* The priority to assign to the task. */
205  NULL ); /* Don't receive a handle back, it is not needed. */
206 
207  xTaskCreate( vRegTest2Task, /* Function that implements the task. */
208  "Reg2", /* Text name of the task. */
209  xRegTestStackSize, /* Stack allocated to the task. */
210  NULL, /* The task parameter is not used. */
211  tskIDLE_PRIORITY, /* The priority to assign to the task. */
212  NULL ); /* Don't receive a handle back, it is not needed. */
213 
214  /* Create the software timer that performs the 'check' functionality,
215  as described at the top of this file. */
216  xCheckTimer = xTimerCreate( "CheckTimer", /* A text name, purely to help debugging. */
217  ( mainCHECK_TIMER_PERIOD_MS ), /* The timer period, in this case 3000ms (3s). */
218  pdTRUE, /* This is an auto-reload timer, so xAutoReload is set to pdTRUE. */
219  ( void * ) 0, /* The ID is not used, so can be set to anything. */
220  prvCheckTimerCallback /* The callback function that inspects the status of all the other tasks. */
221  );
222 
223  /* If the software timer was created successfully, start it. It won't
224  actually start running until the scheduler starts. A block time of
225  zero is used in this call, although any value could be used as the block
226  time will be ignored because the scheduler has not started yet. */
227  if( xCheckTimer != NULL )
228  {
229  xTimerStart( xCheckTimer, mainDONT_BLOCK );
230  }
231 
232  /* Start the kernel. From here on, only tasks and interrupts will run. */
233  vTaskStartScheduler();
234 
235  /* If all is well, the scheduler will now be running, and the following
236  line will never be reached. If the following line does execute, then there
237  was insufficient FreeRTOS heap memory available for the idle and/or timer
238  tasks to be created. See the memory management section on the FreeRTOS web
239  site, or the FreeRTOS tutorial books for more details. */
240  for( ;; );
241 }

244 /* See the description at the top of this file. */
245 static void prvCheckTimerCallback( TimerHandle_t xTimer )
246 {
247 static long lChangedTimerPeriodAlready = pdFALSE;
248 static unsigned long ulLastRegTest1Value = 0, ulLastRegTest2Value = 0;
249 unsigned long ulErrorFound = pdFALSE;
250 
251  /* Check all the demo and test tasks to ensure that they are all still
252  running, and that none have detected an error. */
253  if( xAreIntQueueTasksStillRunning() != pdPASS )
254  {
255  ulErrorFound |= ( 0x01UL << 0UL );
256  }
257 
258  if( xAreBlockTimeTestTasksStillRunning() != pdPASS )
259  {
260  ulErrorFound |= ( 0x01UL << 1UL );
261  }
262 
263  if( xAreCountingSemaphoreTasksStillRunning() != pdPASS )
264  {
265  ulErrorFound |= ( 0x01UL << 2UL );
266  }
267 
268  if( xAreRecursiveMutexTasksStillRunning() != pdPASS )
269  {
270  ulErrorFound |= ( 0x01UL << 3UL );
271  }
272 
273  /* Check that the register test 1 task is still running. */
274  if( ulLastRegTest1Value == ulRegTest1LoopCounter )
275  {
276  ulErrorFound |= ( 0x01UL << 4UL );
277  }
278  ulLastRegTest1Value = ulRegTest1LoopCounter;
279 
280  /* Check that the register test 2 task is still running. */
281  if( ulLastRegTest2Value == ulRegTest2LoopCounter )
282  {
283  ulErrorFound |= ( 0x01UL << 5UL );
284  }
285  ulLastRegTest2Value = ulRegTest2LoopCounter;
286 
287  /* Toggle the check LED to give an indication of the system status. If
288  the LED toggles every mainCHECK_TIMER_PERIOD_MS milliseconds then
289  everything is ok. A faster toggle indicates an error. */
290  vMainToggleLED();
291 
292  /* Have any errors been latched in ulErrorFound? If so, shorten the
293  period of the check timer to mainERROR_CHECK_TIMER_PERIOD_MS milliseconds.
294  This will result in an increase in the rate at which mainCHECK_LED
295  toggles. */
296  if( ulErrorFound != pdFALSE )
297  {
298  if( lChangedTimerPeriodAlready == pdFALSE )
299  {
300  lChangedTimerPeriodAlready = pdTRUE;
301 
302  /* This call to xTimerChangePeriod() uses a zero block time.
303  Functions called from inside of a timer callback function must
304  *never* attempt to block. */
305  xTimerChangePeriod( xTimer, ( mainERROR_CHECK_TIMER_PERIOD_MS ), mainDONT_BLOCK );
306  }
307  }
308 }
