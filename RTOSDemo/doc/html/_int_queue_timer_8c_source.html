90 void vInitialiseTimerForIntQueueTest( void )
91 {
92  /* Enable AHB clock for GPIO and 16-bit timers. */
93  LPC_SYSCON->SYSAHBCLKCTRL |= ( 7 << 6 );
94 
95  /* Interrupt and reset on MR0 match. */
96  LPC_TMR16B0->MCR = 0x03;
97  LPC_TMR16B1->MCR = 0x03;
98 
99  /* Configure the frequency of the interrupt generated by MR0 matches. */
100  LPC_TMR16B0->MR0 = SystemCoreClock / tmrTIMER_2_FREQUENCY;
101  LPC_TMR16B1->MR0 = SystemCoreClock / tmrTIMER_3_FREQUENCY;
102 
103  /* Don't generate interrupts until the scheduler has been started.
104  Interrupts will be automatically enabled when the first task starts
105  running. */
106  taskDISABLE_INTERRUPTS();
107 
108  /* Set the timer interrupts to be above the kernel. The interrupts are
109  assigned different priorities so they nest with each other. */
110  NVIC_SetPriority( TIMER_16_0_IRQn, trmSECOND_HIGHEST_PRIORITY );
111  NVIC_SetPriority( TIMER_16_1_IRQn, tmrMAX_PRIORITY );
112 
113  /* Enable the timer interrupts. */
114  NVIC_EnableIRQ( TIMER_16_0_IRQn );
115  NVIC_EnableIRQ( TIMER_16_1_IRQn );
116 
117  /* Start the timers. */
118  LPC_TMR16B0->TCR = 0x01;
119  LPC_TMR16B1->TCR = 0x01;
120 }
