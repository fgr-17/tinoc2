70 /*
71  The tasks defined on this page demonstrate the use of recursive mutexes.
72 
73  For recursive mutex functionality the created mutex should be created using
74  xSemaphoreCreateRecursiveMutex(), then be manipulated
75  using the xSemaphoreTakeRecursive() and xSemaphoreGiveRecursive() API
76  functions.
77 
78  This demo creates three tasks all of which access the same recursive mutex:
79 
80  prvRecursiveMutexControllingTask() has the highest priority so executes
81  first and grabs the mutex. It then performs some recursive accesses -
82  between each of which it sleeps for a short period to let the lower
83  priority tasks execute. When it has completed its demo functionality
84  it gives the mutex back before suspending itself.
85 
86  prvRecursiveMutexBlockingTask() attempts to access the mutex by performing
87  a blocking 'take'. The blocking task has a lower priority than the
88  controlling task so by the time it executes the mutex has already been
89  taken by the controlling task, causing the blocking task to block. It
90  does not unblock until the controlling task has given the mutex back,
91  and it does not actually run until the controlling task has suspended
92  itself (due to the relative priorities). When it eventually does obtain
93  the mutex all it does is give the mutex back prior to also suspending
94  itself. At this point both the controlling task and the blocking task are
95  suspended.
96 
97  prvRecursiveMutexPollingTask() runs at the idle priority. It spins round
98  a tight loop attempting to obtain the mutex with a non-blocking call. As
99  the lowest priority task it will not successfully obtain the mutex until
100  both the controlling and blocking tasks are suspended. Once it eventually
101  does obtain the mutex it first unsuspends both the controlling task and
102  blocking task prior to giving the mutex back - resulting in the polling
103  task temporarily inheriting the controlling tasks priority.
104 */

171 static void prvRecursiveMutexControllingTask( void *pvParameters )
172 {
173 UBaseType_t ux;
174 
175  /* Just to remove compiler warning. */
176  ( void ) pvParameters;
177 
178  for( ;; )
179  {
180  /* Should not be able to 'give' the mutex, as we have not yet 'taken'
181  it. The first time through, the mutex will not have been used yet,
182  subsequent times through, at this point the mutex will be held by the
183  polling task. */
184  if( xSemaphoreGiveRecursive( xMutex ) == pdPASS )
185  {
186  xErrorOccurred = pdTRUE;
187  }
188 
189  for( ux = 0; ux < recmuMAX_COUNT; ux++ )
190  {
191  /* We should now be able to take the mutex as many times as
192  we like.
193 
194  The first time through the mutex will be immediately available, on
195  subsequent times through the mutex will be held by the polling task
196  at this point and this Take will cause the polling task to inherit
197  the priority of this task. In this case the block time must be
198  long enough to ensure the polling task will execute again before the
199  block time expires. If the block time does expire then the error
200  flag will be set here. */
201  if( xSemaphoreTakeRecursive( xMutex, recmu15ms_DELAY ) != pdPASS )
202  {
203  xErrorOccurred = pdTRUE;
204  }
205 
206  /* Ensure the other task attempting to access the mutex (and the
207  other demo tasks) are able to execute to ensure they either block
208  (where a block time is specified) or return an error (where no
209  block time is specified) as the mutex is held by this task. */
210  vTaskDelay( recmuSHORT_DELAY );
211  }
212 
213  /* For each time we took the mutex, give it back. */
214  for( ux = 0; ux < recmuMAX_COUNT; ux++ )
215  {
216  /* Ensure the other task attempting to access the mutex (and the
217  other demo tasks) are able to execute. */
218  vTaskDelay( recmuSHORT_DELAY );
219 
220  /* We should now be able to give the mutex as many times as we
221  took it. When the mutex is available again the Blocking task
222  should be unblocked but not run because it has a lower priority
223  than this task. The polling task should also not run at this point
224  as it too has a lower priority than this task. */
225  if( xSemaphoreGiveRecursive( xMutex ) != pdPASS )
226  {
227  xErrorOccurred = pdTRUE;
228  }
229 
230  #if( configUSE_PREEMPTION == 0 )
231  taskYIELD();
232  #endif
233  }
234 
235  /* Having given it back the same number of times as it was taken, we
236  should no longer be the mutex owner, so the next give should fail. */
237  if( xSemaphoreGiveRecursive( xMutex ) == pdPASS )
238  {
239  xErrorOccurred = pdTRUE;
240  }
241 
242  /* Keep count of the number of cycles this task has performed so a
243  stall can be detected. */
244  uxControllingCycles++;
245 
246  /* Suspend ourselves so the blocking task can execute. */
247  xControllingIsSuspended = pdTRUE;
248  vTaskSuspend( NULL );
249  xControllingIsSuspended = pdFALSE;
250  }
251 }

254 static void prvRecursiveMutexBlockingTask( void *pvParameters )
255 {
256  /* Just to remove compiler warning. */
257  ( void ) pvParameters;
258 
259  for( ;; )
260  {
261  /* This task will run while the controlling task is blocked, and the
262  controlling task will block only once it has the mutex - therefore
263  this call should block until the controlling task has given up the
264  mutex, and not actually execute past this call until the controlling
265  task is suspended. portMAX_DELAY - 1 is used instead of portMAX_DELAY
266  to ensure the task's state is reported as Blocked and not Suspended in
267  a later call to configASSERT() (within the polling task). */
268  if( xSemaphoreTakeRecursive( xMutex, ( portMAX_DELAY - 1 ) ) == pdPASS )
269  {
270  if( xControllingIsSuspended != pdTRUE )
271  {
272  /* Did not expect to execute until the controlling task was
273  suspended. */
274  xErrorOccurred = pdTRUE;
275  }
276  else
277  {
278  /* Give the mutex back before suspending ourselves to allow
279  the polling task to obtain the mutex. */
280  if( xSemaphoreGiveRecursive( xMutex ) != pdPASS )
281  {
282  xErrorOccurred = pdTRUE;
283  }
284 
285  xBlockingIsSuspended = pdTRUE;
286  vTaskSuspend( NULL );
287  xBlockingIsSuspended = pdFALSE;
288  }
289  }
290  else
291  {
292  /* We should not leave the xSemaphoreTakeRecursive() function
293  until the mutex was obtained. */
294  xErrorOccurred = pdTRUE;
295  }
296 
297  /* The controlling and blocking tasks should be in lock step. */
298  if( uxControllingCycles != ( uxBlockingCycles + 1 ) )
299  {
300  xErrorOccurred = pdTRUE;
301  }
302 
303  /* Keep count of the number of cycles this task has performed so a
304  stall can be detected. */
305  uxBlockingCycles++;
306  }
307 }

310 static void prvRecursiveMutexPollingTask( void *pvParameters )
311 {
312  /* Just to remove compiler warning. */
313  ( void ) pvParameters;
314 
315  for( ;; )
316  {
317  /* Keep attempting to obtain the mutex. We should only obtain it when
318  the blocking task has suspended itself, which in turn should only
319  happen when the controlling task is also suspended. */
320  if( xSemaphoreTakeRecursive( xMutex, recmuNO_DELAY ) == pdPASS )
321  {
322  #if( INCLUDE_eTaskGetState == 1 )
323  {
324  configASSERT( eTaskGetState( xControllingTaskHandle ) == eSuspended );
325  configASSERT( eTaskGetState( xBlockingTaskHandle ) == eSuspended );
326  }
327  #endif /* INCLUDE_eTaskGetState */
328 
329  /* Is the blocking task suspended? */
330  if( ( xBlockingIsSuspended != pdTRUE ) || ( xControllingIsSuspended != pdTRUE ) )
331  {
332  xErrorOccurred = pdTRUE;
333  }
334  else
335  {
336  /* Keep count of the number of cycles this task has performed
337  so a stall can be detected. */
338  uxPollingCycles++;
339 
340  /* We can resume the other tasks here even though they have a
341  higher priority than the polling task. When they execute they
342  will attempt to obtain the mutex but fail because the polling
343  task is still the mutex holder. The polling task (this task)
344  will then inherit the higher priority. The Blocking task will
345  block indefinitely when it attempts to obtain the mutex, the
346  Controlling task will only block for a fixed period and an
347  error will be latched if the polling task has not returned the
348  mutex by the time this fixed period has expired. */
349  vTaskResume( xBlockingTaskHandle );
350  #if( configUSE_PREEMPTION == 0 )
351  taskYIELD();
352  #endif
353 
354  vTaskResume( xControllingTaskHandle );
355  #if( configUSE_PREEMPTION == 0 )
356  taskYIELD();
357  #endif
358 
359  /* The other two tasks should now have executed and no longer
360  be suspended. */
361  if( ( xBlockingIsSuspended == pdTRUE ) || ( xControllingIsSuspended == pdTRUE ) )
362  {
363  xErrorOccurred = pdTRUE;
364  }
365 
366  #if( INCLUDE_uxTaskPriorityGet == 1 )
367  {
368  /* Check priority inherited. */
369  configASSERT( uxTaskPriorityGet( NULL ) == recmuCONTROLLING_TASK_PRIORITY );
370  }
371  #endif /* INCLUDE_uxTaskPriorityGet */
372 
373  #if( INCLUDE_eTaskGetState == 1 )
374  {
375  configASSERT( eTaskGetState( xControllingTaskHandle ) == eBlocked );
376  configASSERT( eTaskGetState( xBlockingTaskHandle ) == eBlocked );
377  }
378  #endif /* INCLUDE_eTaskGetState */
379 
380  /* Release the mutex, disinheriting the higher priority again. */
381  if( xSemaphoreGiveRecursive( xMutex ) != pdPASS )
382  {
383  xErrorOccurred = pdTRUE;
384  }
385 
386  #if( INCLUDE_uxTaskPriorityGet == 1 )
387  {
388  /* Check priority disinherited. */
389  configASSERT( uxTaskPriorityGet( NULL ) == recmuPOLLING_TASK_PRIORITY );
390  }
391  #endif /* INCLUDE_uxTaskPriorityGet */
392  }
393  }
394 
395  #if configUSE_PREEMPTION == 0
396  {
397  taskYIELD();
398  }
399  #endif
400  }
401 }

404 /* This is called to check that all the created tasks are still running. */
405 BaseType_t xAreRecursiveMutexTasksStillRunning( void )
406 {
407 BaseType_t xReturn;
408 static UBaseType_t uxLastControllingCycles = 0, uxLastBlockingCycles = 0, uxLastPollingCycles = 0;
409 
410  /* Is the controlling task still cycling? */
411  if( uxLastControllingCycles == uxControllingCycles )
412  {
413  xErrorOccurred = pdTRUE;
414  }
415  else
416  {
417  uxLastControllingCycles = uxControllingCycles;
418  }
419 
420  /* Is the blocking task still cycling? */
421  if( uxLastBlockingCycles == uxBlockingCycles )
422  {
423  xErrorOccurred = pdTRUE;
424  }
425  else
426  {
427  uxLastBlockingCycles = uxBlockingCycles;
428  }
429 
430  /* Is the polling task still cycling? */
431  if( uxLastPollingCycles == uxPollingCycles )
432  {
433  xErrorOccurred = pdTRUE;
434  }
435  else
436  {
437  uxLastPollingCycles = uxPollingCycles;
438  }
439 
440  if( xErrorOccurred == pdTRUE )
441  {
442  xReturn = pdFAIL;
443  }
444  else
445  {
446  xReturn = pdPASS;
447  }
448 
449  return xReturn;
450 }
