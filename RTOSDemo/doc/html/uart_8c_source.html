179 void UART_Reset (int baudrate)
180 {
181 
182  uint8_t fcr;
183  uint32_t Fdiv;
184  uint32_t regVal;
185 
186  tx_libre = true; // Señalizo el buffer de Tx como vacío
187  UARTCount = 0; // Señalizo la cuenta del buffer en cero
188 
189  /*
190  NVIC_DisableIRQ(UART_IRQn); // Deshabilito la IRQ de la UART
191 
193  // CONFIGURACION DE PINES DE ENTRADAS Y SALIDAS :
195  LPC_IOCON->PIO1_6 &= ~0x07; // UART I/O config
196  LPC_IOCON->PIO1_6 |= 0x01; // UART RXD
197  LPC_IOCON->PIO1_7 &= ~0x07;
198  LPC_IOCON->PIO1_7 |= 0x01; // UART TXD
199 */
201  // CONFIGURACION DE REGISTROS DEL CLOCK :
203  LPC_SYSCON->SYSAHBCLKCTRL |= (1<<12); // Enable UART clock
204  LPC_SYSCON->UARTCLKDIV = 0x1; // divided by 1
205 
206  // LPC_UART->LCR = 0x83;
207  LPC_UART->LCR = U0LCR_WORD_LENGTH_SELECT_8BIT | U0LCR_DLAB; // LCR : Line Control Register : 8 bits, no Parity, 1 Stop bit
208 
209  regVal = LPC_SYSCON->UARTCLKDIV; // Guardo el valor de la división del clock
210 
211  Fdiv = (((SystemCoreClock/LPC_SYSCON->SYSAHBCLKDIV)/regVal)/16)/baudrate ; // Asigno el valor de baud rate
212 
213  LPC_UART->DLM = Fdiv / 256; // DLM : Divisor Latch MSB
214  LPC_UART->DLL = Fdiv % 256; // DLL : Divisor Latch LSB
215 
216  // LPC_UART->LCR = 0x03; // 8N1 y desactivo el DLAB para no tener acceso a DLM y DLL
217  LPC_UART->LCR = U0LCR_WORD_LENGTH_SELECT_8BIT; // Deshabilito el acceso a los registros de división
218 
219  // LPC_UART->FCR = 0x07; // FCR : Habilito y reseteo las FIFO de Tx y Rx
220  fcr = U0FCR_FIFO_ENABLE | // Habilito la fifo y los demas bits
221  U0FCR_RX_FIFO_RESET | // Limpio la fifo de rx
222  U0FCR_TX_FIFO_RESET | // Limpio la fifo de tx
223  U0FCR_RX_TRIGGER_LEVEL_3; // Nivel de trigger 4 bits.
224 
225  LPC_UART->FCR = fcr;
226 
227  regVal = LPC_UART->LSR; // Leo el Line Stauts Register para limpiarlo
228 
229  while ( (LPC_UART->LSR & (LSR_THRE|LSR_TEMT)) != (LSR_THRE|LSR_TEMT) ); // Espero a tener el Buffer de Tx vacío
230 
231  while ( LPC_UART->LSR & LSR_RDR ) // Espero a tener el Buffer Rx vacío
232  {
233  regVal = LPC_UART->RBR; // Leo los bytes pendientes de la fifo Rx
234  }
235 
236  NVIC_EnableIRQ(UART_IRQn); // Vuelvo a habilitar las interrupciones
237 
238  LPC_UART->IER = IER_RBR | IER_THRE | IER_RLS; // IER : Interrupt Enable Register
239 
240  Inicializar_Cola(&cola_rx); // Inicializo cola de recepcion
241  Inicializar_Cola(&cola_tx); // Inicializo cola de transmision
242 
243  Inic_UART_RS485();
244  UART_RS485_Habilitar_Recepcion();
245  return;
