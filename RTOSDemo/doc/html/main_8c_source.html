230 static void prvSetupHardware( void )
231 {
232 extern unsigned long _vStackTop[], _pvHeapStart[];
233 unsigned long ulInterruptStackSize;
234 
235  /* Enable AHB clock for GPIO. */
236  LPC_SYSCON->SYSAHBCLKCTRL |= ( 1 << 6 );
237 
238 
239  /* Configure GPIO for LED output. */
240  // xGPIO0->DIR |= mainLED_BIT;
241 
242  /* The size of the stack used by main and interrupts is not defined in
243  the linker, but just uses whatever RAM is left. Calculate the amount of
244  RAM available for the main/interrupt/system stack, and check it against
245  a reasonable number. If this assert is hit then it is likely you don't
246  have enough stack to start the kernel, or to allow interrupts to nest.
247  Note - this is separate to the stacks that are used by tasks. The stacks
248  that are used by tasks are automatically checked if
249  configCHECK_FOR_STACK_OVERFLOW is not 0 in FreeRTOSConfig.h - but the stack
250  used by interrupts is not. Reducing the conifgTOTAL_HEAP_SIZE setting will
251  increase the stack available to main() and interrupts. */
252  ulInterruptStackSize = ( ( unsigned long ) _vStackTop ) - ( ( unsigned long ) _pvHeapStart );
253  configASSERT( ulInterruptStackSize > 350UL );
254 
255  /* Fill the stack used by main() and interrupts to a known value, so its
256  use can be manually checked. */
257  memcpy( ( void * ) _pvHeapStart, ucExpectedInterruptStackValues, sizeof( ucExpectedInterruptStackValues ) );
258 }
